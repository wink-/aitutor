# Create alias if needed

System Monitoring - htop & iotop

htop - Interactive Process Viewer

# Launch htop
htop

# Sort by CPU usage
htop -s PERCENT_CPU

# Sort by memory usage
htop -s PERCENT_MEM

# Filter by user
htop -u username

# Tree view of processes
htop -t

Key Bindings in htop

  • F1 - Help
  • F2 - Setup
  • F3 - Search
  • F4 - Filter
  • F5 - Tree view
  • F6 - Sort by
  • F9 - Kill process
  • F10 - Quit

iotop - I/O Monitoring

# Launch iotop (requires sudo)
sudo iotop

# Only show processes with I/O activity
sudo iotop -o

# Batch mode (non-interactive)
sudo iotop -b

# Show accumulated I/O
sudo iotop -a

# Show specific processes
sudo iotop -p PID1,PID2

Key Bindings in iotop

  • o - Toggle showing only active processes
  • r - Reverse sort order
  • a - Toggle accumulated mode
  • p - Toggle process/thread mode
  • q - Quit

jq - Command-line JSON processor

Basic Operations

# Pretty print JSON
echo '{"name":"model","version":1}' | jq '.'

# Extract specific field
echo '{"name":"model","version":1}' | jq '.name'

# Get array elements
echo '[1,2,3,4,5]' | jq '.[2]'

# Get array length
echo '[1,2,3,4,5]' | jq 'length'

# Filter arrays
echo '[1,2,3,4,5]' | jq '.[] | select(. > 3)'

Advanced Filtering

# Extract nested fields
jq '.config.model.layers' config.json

# Map over arrays
jq '.models[] | .name' models.json

# Group by field
jq 'group_by(.type)' data.json

# Sort arrays
jq 'sort_by(.score)' results.json

# Combine multiple files
jq -s '.[0] + .[1]' file1.json file2.json

AI API Response Processing

Common jq patterns for AI/ML API responses:

# Extract model predictions
curl -X POST api/predict -d @data.json | jq '.predictions[]'

# Get confidence scores
curl api/models | jq '.[] | select(.accuracy > 0.9) | .name'

# Format training metrics
jq '.epochs[] | {epoch: .number, loss: .train_loss, val_loss: .val_loss}' metrics.json

# Extract hyperparameters
jq '.config | {lr: .learning_rate, batch_size: .batch_size, epochs: .epochs}' config.json

AI Configuration Management

# Parse model configuration
jq '.model.architecture' config.json

# Extract hyperparameters
jq '.training | {lr: .learning_rate, epochs: .epochs, batch_size: .batch_size}' config.json

# Validate configuration
jq 'has("model") and has("training") and has("data")' config.json

# Update configuration
jq '.training.learning_rate = 0.001' config.json > new_config.json

# Compare configurations
jq -s '.[0].training.learning_rate == .[1].training.learning_rate' config1.json config2.json

# Extract dataset info
jq '.data | {train_size: .train.size, val_size: .validation.size, test_size: .test.size}' data_config.json

curl & wget - HTTP Clients for API Testing

curl - Transfer data with URLs

# Basic GET request
curl https://api.example.com/models

# POST request with data
curl -X POST -H "Content-Type: application/json" \
  -d '{"model": "gpt-3"}' \
  https://api.example.com/predict

# Download file
curl -O https://example.com/dataset.zip

# Follow redirects
curl -L https://example.com/redirect

# Save response to file
curl -o response.json https://api.example.com/data

wget - Download files from web

# Download file
wget https://example.com/dataset.zip

# Download recursively
wget -r https://example.com/datasets/

# Continue incomplete downloads
wget -c https://example.com/large_file.tar.gz

# Download with custom user agent
wget --user-agent="AI-Bot/1.0" https://api.example.com/data

# Download in background
wget -b https://example.com/large_dataset.tar.gz

Complete AI Development Workflow

#!/bin/bash
# ai-workflow.sh - Complete AI development monitoring script

# 1. System check
echo "=== System Status ==="
htop -b -n 1 | head -20

# 2. Data validation
echo "=== Data Validation ==="
find data/ -name "*.json" -exec jq empty {} \; 2>&1 | grep -v "parse error" | wc -l

# 3. Configuration review
echo "=== Configuration ==="
bat config.json | jq '.model.architecture, .training.learning_rate'

# 4. Model training
echo "=== Starting Training ==="
python train.py &
TRAIN_PID=$!

# 5. Monitor training process
while kill -0 $TRAIN_PID 2>/dev/null; do
    echo "Training process $TRAIN_PID is running..."
    htop -p $TRAIN_PID -b -n 1 | head -10
    sleep 10
done

# 6. Validate results
echo "=== Training Complete ==="
exa -l --sort=modified models/
bat training.log | tail -10

# 7. Test deployment
echo "=== Testing Deployment ==="
curl -X POST -d @test_input.json localhost:5000/predict | jq '.'

#!/bin/bash
while true; do
    clear
    echo "=== AI Training Dashboard ==="
    echo "System Status:"

    echo "GPU Status:"
    nvidia-smi --query-gpu=temperature.gpu,utilization.gpu,memory.used --format=csv,noheader,nounits
    echo "Recent Logs:"

    sleep 5

# 2. Deployment pipeline
#!/bin/bash
# Deploy model
curl -X POST -F "model=@model.bin" localhost:5000/deploy

# Health check
for i in {1..10}; do
    health=$(curl -s localhost:5000/health | jq -r '.status')
    if [ "$health" == "healthy" ]; then
        echo "Deployment successful"
        break
    fi
    sleep 2
done

# 3. Real-time log analysis
tail -f training.log | while read line; do
    echo "$line" | jq -r 'select(.level == "ERROR") | .message'
done

# 4. Model validation pipeline
for model in models/*.bin; do
    echo "Validating $model"
    curl -X POST -F "model=@$model" -F "data=@test_data.json" \
        localhost:5000/validate | jq '.accuracy'
done

#!/bin/bash

        -d @benchmark_data.json -s localhost:5000/predict > /dev/null

Best Practices for AI Development

Do's
  • Use bat for quick file inspection with syntax highlighting
  • Monitor system resources with htop during training
  • Validate JSON configurations with jq before use
  • Set up aliases for frequently used tool combinations
  • Use iotop to identify I/O bottlenecks in data pipelines
  • Test APIs thoroughly with curl before deployment
  • Create monitoring scripts for long-running processes
Don'ts
  • Don't use cat when bat provides better output
  • Don't ignore system monitoring during resource-intensive tasks
  • Don't parse JSON with text tools when jq is available
  • Don't run iotop without sudo (it needs root privileges)
  • Don't forget to handle errors in API requests
  • Don't use wget for API testing (use curl instead)
  • Don't overwhelm APIs with too many concurrent requests
